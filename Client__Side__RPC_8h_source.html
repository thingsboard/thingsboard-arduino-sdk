48 bool RPC_Request(RPC_Request_Callback const & callback) {
49 char const * method_name = callback.Get_Name();
50
51 if (Helper::stringIsNullorEmpty(method_name)) {
52 Logger::printfln(CLIENT_RPC_METHOD_NULL);
53 return false;
54 }
55 RPC_Request_Callback * registered_callback = nullptr;
56 if (!RPC_Request_Subscribe(callback, registered_callback)) {
57 return false;
58 }
59 else if (registered_callback == nullptr) {
60 return false;
61 }
62
63 JsonArray const * parameters = callback.Get_Parameters();
64
65#if THINGSBOARD_ENABLE_DYNAMIC
66 // String are const char* and therefore stored as a pointer --> zero copy, meaning the size for the strings is 0 bytes,
67 // Data structure size depends on the amount of key value pairs passed + the default method name and params key needed for the request.
68 // See https://arduinojson.org/v6/assistant/ for more information on the needed size for the JsonDocument
69 TBJsonDocument request_buffer(JSON_OBJECT_SIZE(parameters != nullptr ? parameters->size() + 2U : 2U));
70#else
71 // Ensure to have enough size for the infinite amount of possible parameters that could be sent to the cloud
72 StaticJsonDocument<JSON_OBJECT_SIZE(MaxRequestRPC)> request_buffer;
73#endif // THINGSBOARD_ENABLE_DYNAMIC
74
75 request_buffer[RPC_METHOD_KEY] = method_name;
76
77 if (parameters != nullptr && !parameters->isNull()) {
78 request_buffer[RPC_PARAMS_KEY] = *parameters;
79 }
80 else {
81 request_buffer[RPC_PARAMS_KEY] = RPC_EMPTY_PARAMS_VALUE;
82 }
83
84#if !THINGSBOARD_ENABLE_DYNAMIC
85 if (request_buffer.overflowed()) {
86 Logger::printfln(RPC_REQUEST_OVERFLOWED, MaxRequestRPC);
87 return false;
88 }
89#endif // !THINGSBOARD_ENABLE_DYNAMIC
90
91 size_t * p_request_id = m_get_request_id_callback.Call_Callback();
92 if (p_request_id == nullptr) {
93 Logger::printfln(REQUEST_ID_NULL);
94 return false;
95 }
96 auto & request_id = *p_request_id;
97
98 registered_callback->Set_Request_ID(++request_id);
99 registered_callback->Start_Timeout_Timer();
100
101 char topic[Helper::detectSize(RPC_SEND_REQUEST_TOPIC, request_id)] = {};
102 (void)snprintf(topic, sizeof(topic), RPC_SEND_REQUEST_TOPIC, request_id);
103 return m_send_json_callback.Call_Callback(topic, request_buffer, Helper::Measure_Json(request_buffer));
104 }

114 void Process_Json_Response(char const * topic, JsonDocument const & data) override {
115 size_t const request_id = Helper::parseRequestId(RPC_RESPONSE_TOPIC, topic);
116
117#if THINGSBOARD_ENABLE_STL
118 auto it = std::find_if(m_rpc_request_callbacks.begin(), m_rpc_request_callbacks.end(), [&request_id](RPC_Request_Callback & rpc_request) {
119 return rpc_request.Get_Request_ID() == request_id;
120 });
121 if (it != m_rpc_request_callbacks.end()) {
122 auto & rpc_request = *it;
123#else
124 for (auto it = m_rpc_request_callbacks.begin(); it != m_rpc_request_callbacks.end(); ++it) {
125 auto & rpc_request = *it;
126
127 if (rpc_request.Get_Request_ID() != request_id) {
128 continue;
129 }
130#endif // THINGSBOARD_ENABLE_STL
131 rpc_request.Stop_Timeout_Timer();
132 rpc_request.Call_Callback(data);
133
134 // Delete callback because the changes have been requested and the callback is no longer needed
135 Helper::remove(m_rpc_request_callbacks, it);
136#if !THINGSBOARD_ENABLE_STL
137 break;
138#endif // !THINGSBOARD_ENABLE_STL
139 }
140
141 // Attempt to unsubscribe from the shared attribute request topic,
142 // if we are not waiting for any further responses with shared attributes from the server.
143 // Will be resubscribed if another request is sent anyway
144 if (m_rpc_request_callbacks.empty()) {
145 (void)RPC_Request_Unsubscribe();

187 bool RPC_Request_Subscribe(RPC_Request_Callback const & callback, RPC_Request_Callback * & registered_callback) {
188#if !THINGSBOARD_ENABLE_DYNAMIC
189 if (m_rpc_request_callbacks.size() + 1 > m_rpc_request_callbacks.capacity()) {
190 Logger::printfln(MAX_SUBSCRIPTIONS_EXCEEDED, MAX_SUBSCRIPTIONS_TEMPLATE_NAME, CLIENT_SIDE_RPC_SUBSCRIPTIONS);
191 return false;
192 }
193#endif // !THINGSBOARD_ENABLE_DYNAMIC
194 if (!m_subscribe_topic_callback.Call_Callback(RPC_RESPONSE_SUBSCRIBE_TOPIC)) {
195 Logger::printfln(SUBSCRIBE_TOPIC_FAILED, RPC_RESPONSE_SUBSCRIBE_TOPIC);
196 return false;
197 }
198 m_rpc_request_callbacks.push_back(callback);
199 registered_callback = &m_rpc_request_callbacks.back();
200 return true;
201 }

216 // Vectors or array (depends on wheter if THINGSBOARD_ENABLE_DYNAMIC is set to 1 or 0), hold copy of the actual passed data, this is to ensure they stay valid,
217 // even if the user only temporarily created the object before the method was called.
218 // This can be done because all Callback methods mostly consists of pointers to actual object so copying them
219 // does not require a huge memory overhead and is acceptable especially in comparsion to possible problems that could
220 // arise if references were used and the end user does not take care to ensure the Callbacks live on for the entirety
221 // of its usage, which will lead to dangling references and undefined behaviour.
222 // Therefore copy-by-value has been choosen as for this specific use case it is more advantageous,
223 // especially because at most we copy internal vectors or array, that will only ever contain a few pointers
224#if THINGSBOARD_ENABLE_DYNAMIC
225 Vector<RPC_Request_Callback> m_rpc_request_callbacks = {}; // Server side RPC callbacks vector
226#else
227 Array<RPC_Request_Callback, MaxSubscriptions> m_rpc_request_callbacks = {}; // Server side RPC callbacks array
228#endif // THINGSBOARD_ENABLE_DYNAMIC
