271 bool Send_Json(char const * topic, JsonDocument const & source, size_t const & json_size) {
272 // Check if allocating needed memory failed when trying to create the JsonDocument,
273 // if it did the isNull() method will return true. See https://arduinojson.org/v6/api/jsonvariant/isnull/ for more information
274 if (source.isNull()) {
275 Logger::printfln(UNABLE_TO_ALLOCATE_JSON);
276 return false;
277 }
278 // Check if inserting any of the internal values failed because the JsonDocument was too small,
279 // if it did the overflowed() method will return true. See https://arduinojson.org/v6/api/jsondocument/overflowed/ for more information
280 if (source.overflowed()) {
281 Logger::printfln(JSON_SIZE_TO_SMALL);
282 return false;
283 }
284 bool result = false;
285
286#if THINGSBOARD_ENABLE_STREAM_UTILS
287 // Check if the size of the given message would be too big for the actual client,
288 // if it is utilize the serialize json work around, so that the internal client buffer can be circumvented
289 if (m_client.get_buffer_size() < json_size) {
290#if THINGSBOARD_ENABLE_DEBUG
291 Logger::printfln(SEND_MESSAGE, topic, SEND_SERIALIZED);
292#endif // THINGSBOARD_ENABLE_DEBUG
293 result = Serialize_Json(topic, source, json_size - 1);
294 }
295 // Check if the remaining stack size of the current task would overflow the stack,
296 // if it would allocate the memory on the heap instead to ensure no stack overflow occurs
297 else
298#endif // THINGSBOARD_ENABLE_STREAM_UTILS
299 if (json_size > getMaximumStackSize()) {
300 char* json = new char[json_size]();
301 if (serializeJson(source, json, json_size) < json_size - 1) {
302 Logger::printfln(UNABLE_TO_SERIALIZE_JSON);
303 }
304 else {
305 result = Send_Json_String(topic, json);
306 }
307 // Ensure to actually delete the memory placed onto the heap, to make sure we do not create a memory leak
308 // and set the pointer to null so we do not have a dangling reference.
309 delete[] json;
310 json = nullptr;
311 }
312 else {
313 char json[json_size] = {};
314 if (serializeJson(source, json, json_size) < json_size - 1) {
315 Logger::printfln(UNABLE_TO_SERIALIZE_JSON);
316 return result;
317 }
318 result = Send_Json_String(topic, json);
319 }
320
321 return result;
322 }

328 bool Send_Json_String(char const * topic, char const * json) {
329 if (json == nullptr) {
330 return false;
331 }
332
333 uint16_t current_send_buffer_size = m_client.get_send_buffer_size();
334 size_t const json_size = strlen(json);
335
336 if (current_send_buffer_size < json_size) {

343#endif // THINGSBOARD_ENABLE_DEBUG
344 return m_client.publish(topic, reinterpret_cast<uint8_t const *>(json), json_size);
345 }

639 bool sendKeyValue(char const * key, T const & value, bool telemetry = true) {
640 const Telemetry t(key, value);
641 if (t.IsEmpty()) {
642 return false;
643 }
644
645 StaticJsonDocument<JSON_OBJECT_SIZE(1)> json_buffer;
646 if (!t.SerializeKeyValue(json_buffer)) {
647 Logger::printfln(UNABLE_TO_SERIALIZE);
648 return false;
649 }

668 bool sendDataArray(InputIterator const & first, InputIterator const & last, bool telemetry) {
669 size_t const size = Helper::distance(first, last);
670#if THINGSBOARD_ENABLE_DYNAMIC
671 // char const * are stored as only a pointer inside the JsonDocument --> zero copy, meaning the size for the strings is 0 bytes.
672 // Data structure size, therefore only depends on the amount of key value pairs passed.
673 // See https://arduinojson.org/v6/assistant/ for more information on the needed size for the JsonDocument
674 TBJsonDocument json_buffer(JSON_OBJECT_SIZE(size));
675#else
676 if (size > MaxKeyValuePairAmount) {
677 Logger::printfln(TOO_MANY_JSON_FIELDS, size, "MaxKeyValuePairAmount", MaxKeyValuePairAmount);
678 return false;
679 }
680 StaticJsonDocument<JSON_OBJECT_SIZE(MaxKeyValuePairAmount)> json_buffer;
681#endif // THINGSBOARD_ENABLE_DYNAMIC
682
683#if THINGSBOARD_ENABLE_STL
684 if (std::any_of(first, last, [&json_buffer](Telemetry const & data) { return !data.SerializeKeyValue(json_buffer); })) {
685 Logger::printfln(UNABLE_TO_SERIALIZE);
686 return false;
687 }
688#else
689 for (auto it = first; it != last; ++it) {
690 auto const & data = *it;
691 if (!data.SerializeKeyValue(json_buffer)) {
692 Logger::printfln(UNABLE_TO_SERIALIZE);
693 return false;
694 }
695 }
696#endif // THINGSBOARD_ENABLE_STL
697 return telemetry ? sendTelemetryJson(json_buffer, Helper::Measure_Json(json_buffer)) : sendAttributeJson(json_buffer, Helper::Measure_Json(json_buffer));
698 }

710 void onMQTTMessage(char * topic, uint8_t * payload, unsigned int length) {
711#if THINGSBOARD_ENABLE_DEBUG
712 Logger::printfln(RECEIVE_MESSAGE, length, topic);
713#endif // THINGSBOARD_ENABLE_DEBUG
714
715#if THINGSBOARD_ENABLE_STL
716#if THINGSBOARD_ENABLE_CXX20
717 auto filtered_raw_api_implementations = m_api_implementations | std::views::filter([&topic](IAPI_Implementation const * api) {
718#else
719#if THINGSBOARD_ENABLE_DYNAMIC
720 Vector<IAPI_Implementation *> filtered_raw_api_implementations = {};
721#else
722 Array<IAPI_Implementation *, MaxEndpointsAmount> filtered_raw_api_implementations = {};
723#endif // THINGSBOARD_ENABLE_DYNAMIC
724 std::copy_if(m_api_implementations.begin(), m_api_implementations.end(), std::back_inserter(filtered_raw_api_implementations), [&topic](IAPI_Implementation const * api) {
725#endif // THINGSBOARD_ENABLE_CXX20
726 return (api != nullptr && api->Get_Process_Type() == API_Process_Type::RAW && api->Compare_Response_Topic(topic));
727 });
728
729 for (auto & api : filtered_raw_api_implementations) {
730 api->Process_Response(topic, payload, length);
731 }
732
733 // If the filtered api implementations was not emtpy it means the response was processed as its raw bytes representation atleast once,
734 // and because we interpreted it as raw bytes instead of json, we skip the further processing of those raw bytes as json.
735 // We do that because the received response is in that case not even valid json in the first place and would therefore simply fail deserialization
736 if (!filtered_raw_api_implementations.empty()) {
737 return;
738 }
739#else
740 bool processed_response_as_raw = false;
741 for (auto & api : m_api_implementations) {
742 if (api == nullptr || api->Get_Process_Type() != API_Process_Type::RAW || !api->Compare_Response_Topic(topic)) {
743 continue;
744 }
745 api->Process_Response(topic, payload, length);
746 processed_response_as_raw = true;
747 }
748
749 if (processed_response_as_raw) {
750 return;
751 }
752#endif // THINGSBOARD_ENABLE_STL
753
754 // Calculate size with the total amount of commas, always denotes the end of a key-value pair besides for the last element in an array or in an object where the comma is not permitted,
755 // therfore we have to add the space for another key-value pair for all the occurences of thoose symbols as well
756 size_t const size = Helper::getOccurences(payload, ',', length) + Helper::getOccurences(payload, '{', length) + Helper::getOccurences(payload, '[', length);
757#if THINGSBOARD_ENABLE_DYNAMIC
758 // Buffer that we deserialize is writeable and not read only and therefore stored as a pointer inside the JsonDocument --> zero copy, meaning the size for the received payload is 0 bytes.
759 // Data structure size, therefore only depends on the amount of key value pairs received.
760 // See https://arduinojson.org/v6/assistant/ for more information on the needed size for the JsonDocument
761 size_t const document_size = JSON_OBJECT_SIZE(size);
762 if (m_max_response_size != 0U && document_size > m_max_response_size) {
763 Logger::printfln(MAXIMUM_RESPONSE_EXCEEDED, document_size, m_max_response_size);
764 return;
765 }
766 TBJsonDocument json_buffer(document_size);
767 // Because we calcualte the allocation dynamically fromt he payload, which is user input, it could theoretically be malicious ({ "malicious" : "{{{{{{{{{..."}) and contain a lot of the symbols used to calculate the size.
768 // But if that is the case adn the allocation still succeeds we delete the allocated memory relatively fast again so it shouldn't be a problem and if the allocation fails we simply return at this point with an appropriate error message
769 if (json_buffer.capacity() != document_size) {
770 Logger::printfln(HEAP_ALLOCATION_FAILED, document_size);
771 return;
772 }
773#else
774 if (size > MaxResponse) {
775 Logger::printfln(TOO_MANY_JSON_FIELDS, size, "MaxResponse", MaxResponse);
776 return;
777 }
778 size_t const document_size = JSON_OBJECT_SIZE(MaxResponse);
779 StaticJsonDocument<document_size> json_buffer;
780#endif // THINGSBOARD_ENABLE_DYNAMIC
781#if THINGSBOARD_ENABLE_DEBUG
782 Logger::printfln(ALLOCATING_JSON, document_size);
783#endif // THINGSBOARD_ENABLE_DEBUG
784
785 // The deserializeJson method we use, can use the zero copy mode because a writeable input was passed,
786 // if that were not the case the needed allocated memory would drastically increase, because the keys would need to be copied as well.
787 // See https://arduinojson.org/v6/doc/deserialization/ for more info on ArduinoJson deserialization
788 DeserializationError const error = deserializeJson(json_buffer, payload, length);
789 if (error) {
790 Logger::printfln(UNABLE_TO_DE_SERIALIZE_JSON, error.c_str());
791 return;
792 }
