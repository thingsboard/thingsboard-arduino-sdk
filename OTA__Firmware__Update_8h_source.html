237 bool Prepare_Firmware_Settings(OTA_Update_Callback const & callback) {
238 char const * current_fw_title = callback.Get_Firmware_Title();
239 char const * current_fw_version = callback.Get_Firmware_Version();
240
241 if (Helper::stringIsNullorEmpty(current_fw_title) || Helper::stringIsNullorEmpty(current_fw_version)) {
242 return false;
243 }
244 else if (!Firmware_Send_Info(current_fw_title, current_fw_version)) {
245 return false;
246 }
247
248 size_t * p_request_id = m_get_request_id_callback.Call_Callback();
249 if (p_request_id == nullptr) {
250 Logger::printfln(REQUEST_ID_NULL);
251 return false;
252 }
253 auto & request_id = *p_request_id;
254
255 m_fw_callback = callback;
256 m_fw_callback.Set_Request_ID(++request_id);
257 (void)snprintf(m_response_topic, sizeof(m_response_topic), FIRMWARE_RESPONSE_TOPIC, request_id);
258 return true;
259 }

277 bool Firmware_OTA_Unsubscribe() {
278 // Buffer size has been set to another value before the update,
279 // to allow to receive ota chunck packets that might be much bigger than the normal
280 // buffer size would allow, therefore we return to the previous value to decrease overall memory usage
281 if (m_changed_buffer_size) {

318 void Firmware_Shared_Attribute_Received(JsonObjectConst const & data) {
319 // Check if firmware is available for our device
320 if (!data.containsKey(FW_VER_KEY) || !data.containsKey(FW_TITLE_KEY) || !data.containsKey(FW_CHKS_KEY) || !data.containsKey(FW_CHKS_ALGO_KEY) || !data.containsKey(FW_SIZE_KEY)) {
321 Logger::printfln(NO_FW);
322 Firmware_Send_State(FW_STATE_FAILED, NO_FW);
323 return;
324 }
325
326 char const * fw_title = data[FW_TITLE_KEY];
327 char const * fw_version = data[FW_VER_KEY];
328 char const * fw_checksum = data[FW_CHKS_KEY];
329 char const * fw_algorithm = data[FW_CHKS_ALGO_KEY];
330 size_t const fw_size = data[FW_SIZE_KEY];
331
332 char const * curr_fw_title = m_fw_callback.Get_Firmware_Title();
333 char const * curr_fw_version = m_fw_callback.Get_Firmware_Version();
334
335 if (fw_title == nullptr || fw_version == nullptr || curr_fw_title == nullptr || curr_fw_version == nullptr || fw_algorithm == nullptr || fw_checksum == nullptr) {
336 Logger::printfln(EMPTY_FW);
337 Firmware_Send_State(FW_STATE_FAILED, EMPTY_FW);
338 return;
339 }
340 // If firmware version and title is the same, we do not initiate an update, because we expect the type of binary to be the same one we are currently using
341 // and therefore updating would be useless as we have already updated previously
342 else if (strncmp(curr_fw_title, fw_title, strlen(curr_fw_title)) == 0 && strncmp(curr_fw_version, fw_version, strlen(curr_fw_version)) == 0) {
343 Firmware_Send_State(FW_STATE_UPDATED);
344 return;
345 }
346 // If firmware title is not the same, we do not initiate an update, because we expect the binary to be for another type of device
347 // and downloading it on this device could possibly cause hardware issues or even destroy the device
348 else if (strncmp(curr_fw_title, fw_title, strlen(curr_fw_title)) != 0) {
349 char message[strlen(FW_NOT_FOR_US) + strlen(fw_title) + strlen(curr_fw_title) + 3] = {};
350 (void)snprintf(message, sizeof(message), FW_NOT_FOR_US, fw_title, curr_fw_title);
351 Logger::printfln(message);
352 Firmware_Send_State(FW_STATE_FAILED, message);
353 return;
354 }
355
356 mbedtls_md_type_t fw_checksum_algorithm = mbedtls_md_type_t{};
357
358 if (strncmp(CHECKSUM_AGORITM_MD5, fw_algorithm, strlen(CHECKSUM_AGORITM_MD5)) == 0) {
359 fw_checksum_algorithm = mbedtls_md_type_t::MBEDTLS_MD_MD5;
360 }
361 else if (strncmp(CHECKSUM_AGORITM_SHA256, fw_algorithm, strlen(CHECKSUM_AGORITM_SHA256)) == 0) {
362 fw_checksum_algorithm = mbedtls_md_type_t::MBEDTLS_MD_SHA256;
363 }
364 else if (strncmp(CHECKSUM_AGORITM_SHA384, fw_algorithm, strlen(CHECKSUM_AGORITM_SHA384)) == 0) {
365 fw_checksum_algorithm = mbedtls_md_type_t::MBEDTLS_MD_SHA384;
366 }
367 else if (strncmp(CHECKSUM_AGORITM_SHA512, fw_algorithm, strlen(CHECKSUM_AGORITM_SHA512)) == 0) {
368 fw_checksum_algorithm = mbedtls_md_type_t::MBEDTLS_MD_SHA512;
369 }
370 else {
371 char message[strlen(FW_CHKS_ALGO_NOT_SUPPORTED) + strlen(fw_algorithm) + 2] = {};
372 (void)snprintf(message, sizeof(message), FW_CHKS_ALGO_NOT_SUPPORTED, fw_algorithm);
373 Logger::printfln(message);
374 Firmware_Send_State(FW_STATE_FAILED, message);
375 return;
376 }
377
378 m_fw_callback.Call_Update_Starting_Callback();
379 bool const result = Firmware_OTA_Subscribe();
380 if (!result) {
381 m_fw_callback.Call_Callback(result);
382 return;
383 }
384
385#if THINGSBOARD_ENABLE_DEBUG
386 Logger::printfln(PAGE_BREAK);
387 Logger::printfln(NEW_FW);
388 char firmware[strlen(FROM_TOO) + strlen(curr_fw_version) + strlen(fw_version) + 3] = {};
389 (void)snprintf(firmware, sizeof(firmware), FROM_TOO, curr_fw_version, fw_version);
390 Logger::printfln(firmware);
391 Logger::printfln(DOWNLOADING_FW);
392#endif // THINGSBOARD_ENABLE_DEBUG
393
394 // Calculate the number of chuncks we need to request,
395 // in order to download the complete firmware binary
396 const uint16_t& chunk_size = m_fw_callback.Get_Chunk_Size();
397
398 // Get the previous buffer size and cache it so the previous settings can be restored.
399 m_previous_buffer_size = m_get_receive_size_callback.Call_Callback();
400 m_changed_buffer_size = m_previous_buffer_size < (chunk_size + 50U);
401
402 // Increase size of receive buffer
403 if (m_changed_buffer_size && !m_set_buffer_size_callback.Call_Callback(chunk_size + 50U, m_get_send_size_callback.Call_Callback())) {
404 Logger::printfln(NOT_ENOUGH_RAM);
405 Firmware_Send_State(FW_STATE_FAILED, NOT_ENOUGH_RAM);
406 m_fw_callback.Call_Callback(false);
407 return;
408 }
409
410 m_ota.Start_Firmware_Update(m_fw_callback, fw_size, fw_checksum, fw_checksum_algorithm);
411 }
