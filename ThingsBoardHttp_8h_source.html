71 bool Send_Json(char const * topic, JsonDocument const & source, size_t const & json_size) {
72 // Check if allocating needed memory failed when trying to create the JsonDocument,
73 // if it did the isNull() method will return true. See https://arduinojson.org/v6/api/jsonvariant/isnull/ for more information
74 if (source.isNull()) {
75 Logger::printfln(UNABLE_TO_ALLOCATE_JSON);
76 return false;
77 }
78 // Check if inserting any of the internal values failed because the JsonDocument was too small,
79 // if it did the overflowed() method will return true. See https://arduinojson.org/v6/api/jsondocument/overflowed/ for more information
80 if (source.overflowed()) {
81 Logger::printfln(JSON_SIZE_TO_SMALL);
82 return false;
83 }
84 bool result = false;
85 if (getMaximumStackSize() < json_size) {
86 char * json = new char[json_size]();
87 if (serializeJson(source, json, json_size) < json_size - 1) {
88 Logger::printfln(UNABLE_TO_SERIALIZE_JSON);
89 }
90 else {
91 result = Send_Json_String(topic, json);
92 }
93 // Ensure to actually delete the memory placed onto the heap, to make sure we do not create a memory leak
94 // and set the pointer to null so we do not have a dangling reference.
95 delete[] json;
96 json = nullptr;
97 }
98 else {
99 char json[json_size] = {};
100 if (serializeJson(source, json, json_size) < json_size - 1) {
101 Logger::printfln(UNABLE_TO_SERIALIZE_JSON);
102 return result;
103 }
104 result = Send_Json_String(topic, json);
105 }
106 return result;
107 }

113 bool Send_Json_String(char const * topic, char const * json) {
114 if (json == nullptr || m_token == nullptr) {
115 return false;
116 }
117
118 char path[Helper::detectSize(topic, m_token)] = {};
119 (void)snprintf(path, sizeof(path), topic, m_token);
120 return postMessage(path, json);
121 }

271 bool postMessage(char const * path, char const * json) {
272 bool success = m_client.post(path, HTTP_POST_PATH, json) == 0;
273 int const status = m_client.get_response_status_code();
274
275 if (!success || status < HTTP_RESPONSE_SUCCESS_RANGE_START || status > HTTP_RESPONSE_SUCCESS_RANGE_END) {
276 Logger::printfln(HTTP_FAILED, POST, status);
277 success = false;
278 }
279
280 clearConnection();
281 return success;
282 }

289#if THINGSBOARD_ENABLE_STL
290 bool getMessage(char const * path, std::string& response) {
291#else
292 bool getMessage(char const * path, String& response) {
293#endif // THINGSBOARD_ENABLE_STL
294 bool success = m_client.get(path);
295 int const status = m_client.get_response_status_code();
296
297 if (!success || status < HTTP_RESPONSE_SUCCESS_RANGE_START || status > HTTP_RESPONSE_SUCCESS_RANGE_END) {
298 Logger::printfln(HTTP_FAILED, GET, status);
299 success = false;
300 goto cleanup;
301 }
302 response = m_client.get_response_body();
303
304 cleanup:
305 clearConnection();
306 return success;
307 }

324 bool sendDataArray(InputIterator const & first, InputIterator const & last, bool telemetry) {
325 size_t const size = Helper::distance(first, last);
326#if THINGSBOARD_ENABLE_DYNAMIC
327 TBJsonDocument json_buffer(JSON_OBJECT_SIZE(size));
328#else
329 if (size > MaxKeyValuePairAmount) {
330 Logger::printfln(TOO_MANY_JSON_FIELDS, size, "MaxKeyValuePairAmount", MaxKeyValuePairAmount);
331 return false;
332 }
333 StaticJsonDocument<JSON_OBJECT_SIZE(MaxKeyValuePairAmount)> json_buffer;
334#endif // THINGSBOARD_ENABLE_DYNAMIC
335
336#if THINGSBOARD_ENABLE_STL
337 if (std::any_of(first, last, [&json_buffer](Telemetry const & data) { return !data.SerializeKeyValue(json_buffer); })) {
338 Logger::printfln(UNABLE_TO_SERIALIZE);
339 return false;
340 }
341#else
342 for (auto it = first; it != last; ++it) {
343 auto const & data = *it;
344 if (!data.SerializeKeyValue(json_buffer)) {
345 Logger::printfln(UNABLE_TO_SERIALIZE);
346 return false;
347 }
348 }
349#endif // THINGSBOARD_ENABLE_STL
350 return telemetry ? sendTelemetryJson(json_buffer, Helper::Measure_Json(json_buffer)) : sendAttributeJson(json_buffer, Helper::Measure_Json(json_buffer));
351 }

359 template<typename T>
360 bool sendKeyValue(char const * key, T value, bool telemetry = true) {
361 Telemetry const t(key, value);
362 if (t.IsEmpty()) {
363 // Message is ignored and not sent at all.
364 return false;
365 }
366
367 StaticJsonDocument<JSON_OBJECT_SIZE(1)> json_buffer;
368 if (!t.SerializeKeyValue(json_buffer)) {
369 Logger::printfln(UNABLE_TO_SERIALIZE);
370 return false;
371 }
372 return telemetry ? sendTelemetryJson(json_buffer, Helper::Measure_Json(json_buffer)) : sendAttributeJson(json_buffer, Helper::Measure_Json(json_buffer));
373 }
