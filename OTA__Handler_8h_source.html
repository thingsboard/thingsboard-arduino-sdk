105 void Process_Firmware_Packet(size_t const & current_chunk, uint8_t * payload, size_t const & total_bytes) {
106 if (current_chunk != m_requested_chunks) {
107 Logger::printfln(RECEIVED_UNEXPECTED_CHUNK, current_chunk, m_requested_chunks);
108 return;
109 }
110 size_t expected_chunk_size = 0U;
111 if (!Received_Valid_Chunk_Size(total_bytes, expected_chunk_size)) {
112 Logger::printfln(RECEIVED_UNEXPECTED_CHUNK_SIZE, expected_chunk_size, total_bytes);
113 return;
114 }
115
116 m_watchdog.detach();
117 #if THINGSBOARD_ENABLE_DEBUG
118 Logger::printfln(FW_CHUNK, current_chunk, total_bytes);
119 #endif // THINGSBOARD_ENABLE_DEBUG
120
121 if (current_chunk == 0U) {
122 // Initialize Flash
123 if (!m_fw_updater->begin(m_fw_size)) {
124 Logger::printfln(ERROR_UPDATE_BEGIN);
125 return Handle_Failure(OTA_Failure_Response::RETRY_UPDATE, ERROR_UPDATE_BEGIN);
126 }
127 }
128
129 // Write received binary data to flash partition
130 size_t const written_bytes = m_fw_updater->write(payload, total_bytes);
131 if (written_bytes != total_bytes) {
132 char message[Helper::detectSize(ERROR_UPDATE_WRITE, written_bytes, total_bytes)] = {};
133 (void)snprintf(message, sizeof(message), ERROR_UPDATE_WRITE, written_bytes, total_bytes);
134 Logger::printfln(message);
135 return Handle_Failure(OTA_Failure_Response::RETRY_UPDATE, message);
136 }
137
138 // Update value only if writing to flash was a success, result is ignored,
139 // because it can only fail if the input parameters are invalid
140 (void)m_hash.update(payload, total_bytes);
141
142 m_requested_chunks = current_chunk + 1;
143 m_fw_callback->Call_Progress_Callback(m_requested_chunks, m_total_chunks);
144
145 // Ensure to check if the update was cancelled during the progress callback,
146 // if it was the callback variable was reset and there is no need to request the next firmware packet
147 if (m_fw_callback == nullptr) {
148 Logger::printfln(OTA_CB_IS_NULL);
149 return Handle_Failure(OTA_Failure_Response::RETRY_NOTHING, OTA_CB_IS_NULL);
150 }
151
152 // Reset retries as the current chunk has been downloaded and handled successfully
153 m_retries = m_fw_callback->Get_Chunk_Retries();
154 Request_Next_Firmware_Packet();
155 }

172 bool Received_Valid_Chunk_Size(size_t const & received_chunk_size, size_t & expected_chunk_size) {
173 bool const is_last_chunk = m_requested_chunks + 1 >= m_total_chunks;
174 if (is_last_chunk) {
175 size_t const last_chunk_expected_size = m_fw_size % m_fw_callback->Get_Chunk_Size();
176 expected_chunk_size = last_chunk_expected_size;
177 return received_chunk_size == last_chunk_expected_size;
178 }
179 expected_chunk_size = m_fw_callback->Get_Chunk_Size();
180 return received_chunk_size == m_fw_callback->Get_Chunk_Size();
181 }

184 void Request_First_Firmware_Packet() {
185 m_requested_chunks = 0U;
186 m_retries = m_fw_callback->Get_Chunk_Retries();
187 // Hash start result is ignored, because it can only fail if the input parameters are invalid
188 (void)m_hash.start(m_fw_checksum_algorithm);

192 }

196 void Request_Next_Firmware_Packet() {
197 // Check if we have already requested and handled the last remaining chunk
198 if (m_requested_chunks >= m_total_chunks) {
199 Finish_Firmware_Update();
200 return;
201 }
202
203 if (!m_publish_callback.Call_Callback(m_fw_callback->Get_Request_ID(), m_requested_chunks)) {
204 Logger::printfln(UNABLE_TO_REQUEST_CHUNCKS);
205 }
206
207 // Watchdog gets started no matter if publishing request was successful or not in hopes,
208 // that after the given timeout the callback calls this method again and can then publish the request successfully.
209 // This works because the request fails most of the time, because the internet connection might have been temporarily disconnected.
210 // Therefore waiting a while and then retrying, means we might be reconnected again
211 m_watchdog.once(m_fw_callback->Get_Timeout());
212 }

217 void Finish_Firmware_Update() {
218 (void)m_send_fw_state_callback.Call_Callback(FW_STATE_DOWNLOADED, "");
219
220 char calculated_checksum[FIRMWARE_HASH_SIZE] = {};
221 // Result of calculating final hash result is ignored,
222 // because it can only fail if the input parameters are invalid and we check it afterwards anyway
223 (void)m_hash.finish(calculated_checksum);
224
225 if (strncmp(m_fw_checksum, calculated_checksum, strlen(m_fw_checksum)) != 0) {
226 char message[Helper::detectSize(CHECKSUM_VERIFICATION_FAILED, calculated_checksum, m_fw_checksum)] = {};
227 (void)snprintf(message, sizeof(message), CHECKSUM_VERIFICATION_FAILED, calculated_checksum, m_fw_checksum);
228 Logger::printfln(message);
229 return Handle_Failure(OTA_Failure_Response::RETRY_UPDATE, message);
230 }
231
232 #if THINGSBOARD_ENABLE_DEBUG
233 Logger::printfln(CHECKSUM_VERIFICATION_SUCCESS);
234 #endif // THINGSBOARD_ENABLE_DEBUG
235
236 if (!m_fw_updater->end()) {
237 Logger::printfln(ERROR_UPDATE_END);
238 return Handle_Failure(OTA_Failure_Response::RETRY_UPDATE, ERROR_UPDATE_END);
239 }
240
241 #if THINGSBOARD_ENABLE_DEBUG
242 Logger::printfln(FW_UPDATE_SUCCESS);
243 #endif // THINGSBOARD_ENABLE_DEBUG
244
245 (void)m_send_fw_state_callback.Call_Callback(FW_STATE_UPDATING, "");
246 m_fw_callback->Call_Callback(true);
247 (void)m_finish_callback.Call_Callback();
248 }

254 void Handle_Failure(OTA_Failure_Response const & failure_response, char const * error_message) {
255 if (m_retries <= 0) {
256 (void)m_send_fw_state_callback.Call_Callback(FW_STATE_FAILED, error_message);
257 m_fw_callback->Call_Callback(false);
258 (void)m_finish_callback.Call_Callback();
259 return;
260 }
261
262 // Decrease the amount of retries of downloads for the current chunk,
263 // reset as soon as the next chunk has been received and handled successfully
264 m_retries--;
265
266 switch (failure_response) {
267 case OTA_Failure_Response::RETRY_CHUNK:
268 Request_Next_Firmware_Packet();
269 break;
270 case OTA_Failure_Response::RETRY_UPDATE:
271 Request_First_Firmware_Packet();
272 break;
273 case OTA_Failure_Response::RETRY_NOTHING:
274 (void)m_send_fw_state_callback.Call_Callback(FW_STATE_FAILED, error_message);
275 m_fw_callback->Call_Callback(false);
276 (void)m_finish_callback.Call_Callback();
277 break;
278 default:
279 // Nothing to do
280 break;
281 }
282 }
