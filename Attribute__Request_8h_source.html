180 auto const & attributes = callback.Get_Attributes();
181
182 // Check if any sharedKeys were requested
183 if (attributes.empty()) {
184#if THINGSBOARD_ENABLE_DEBUG

186#endif // THINGSBOARD_ENABLE_DEBUG

188 }

198#else
199 Attribute_Request_Callback<MaxAttributes> * registered_callback = nullptr;
200#endif // THINGSBOARD_ENABLE_DYNAMIC
201 if (!Attributes_Request_Subscribe(callback, registered_callback)) {

208 // String are const char* and therefore stored as a pointer --> zero copy, meaning the size for the strings is 0 bytes,
209 // Data structure size depends on the amount of key value pairs passed + the default clientKeys or sharedKeys
210 // See https://arduinojson.org/v6/assistant/ for more information on the needed size for the JsonDocument
211 StaticJsonDocument<JSON_OBJECT_SIZE(1)> request_buffer;
212
213 // Calculate the size required for the char buffer containing all the attributes seperated by a comma,
214 // before initalizing it so it is possible to allocate it on the stack
215 size_t size = 0U;
216 for (const auto & att : attributes) {
217 if (Helper::stringIsNullorEmpty(att)) {
218 continue;
219 }
220
221 size += strlen(att);
222 size += strlen(",");
223 }
224
225 // Initalizes complete array to 0, required because strncat needs both destination and source to contain proper null terminated strings
226 char request[size] = {};
227 for (const auto & att : attributes) {
228 if (Helper::stringIsNullorEmpty(att)) {
229#if THINGSBOARD_ENABLE_DEBUG
230 Logger::printfln(ATT_KEY_IS_NULL);
231#endif // THINGSBOARD_ENABLE_DEBUG
232 continue;
233 }
234
235 strncat(request, att, size);
236 size -= strlen(att);
237 strncat(request, ",", size);
238 size -= strlen(",");
239 }
240
241 // Ensure to cast to const, this is done so that ArduinoJson does not copy the value but instead simply store the pointer, which does not require any more memory,
242 // besides the base size needed to allocate one key-value pair. Because if we don't the char array would be copied
243 // and because there is not enough space the value would simply be "undefined" instead. Which would cause the request to not be sent correctly
244 request_buffer[attribute_request_key] = static_cast<const char*>(request);
245
246 size_t * p_request_id = m_get_request_id_callback.Call_Callback();
247 if (p_request_id == nullptr) {
248 Logger::printfln(REQUEST_ID_NULL);
249 return false;
250 }
251 auto & request_id = *p_request_id;
252
253 registered_callback->Set_Request_ID(++request_id);
254 registered_callback->Set_Attribute_Key(attribute_response_key);
255 registered_callback->Start_Timeout_Timer();
256
257 char topic[Helper::detectSize(ATTRIBUTE_REQUEST_TOPIC, request_id)] = {};
258 (void)snprintf(topic, sizeof(topic), ATTRIBUTE_REQUEST_TOPIC, request_id);
259 return m_send_json_callback.Call_Callback(topic, request_buffer, Helper::Measure_Json(request_buffer));
260 }

299 // Vectors or array (depends on wheter if THINGSBOARD_ENABLE_DYNAMIC is set to 1 or 0), hold copy of the actual passed data, this is to ensure they stay valid,
300 // even if the user only temporarily created the object before the method was called.
301 // This can be done because all Callback methods mostly consists of pointers to actual object so copying them
302 // does not require a huge memory overhead and is acceptable especially in comparsion to possible problems that could
303 // arise if references were used and the end user does not take care to ensure the Callbacks live on for the entirety
304 // of its usage, which will lead to dangling references and undefined behaviour.
305 // Therefore copy-by-value has been choosen as for this specific use case it is more advantageous,
306 // especially because at most we copy internal vectors or array, that will only ever contain a few pointers
