230 // Calls esp_mqtt_set_config(), which should adjust the underlying mqtt client to the changed values.
231 // Which it does but not for the buffer_size, this results in the buffer size only being able to be changed when initally creating the mqtt client.
232 // If the mqtt client is reinitalized this causes disconnected and reconnects tough and the connection becomes unstable.
233 // Therefore this workaround can also not be used. Instead we expect the esp_mqtt_set_config(), to do what the name implies and therefore still call it
234 // and created an issue revolving around the aformentioned problem so it might get fixed in future version of the esp_mqtt client.
235 // See https://github.com/espressif/esp-mqtt/issues/267 for more information on the issue
236 return update_configuration();

278 bool connect(char const * client_id, char const * user_name, char const * password) override {
279#if ESP_IDF_VERSION_MAJOR < 5
280 m_mqtt_configuration.client_id = client_id;
281 m_mqtt_configuration.username = user_name;
282 m_mqtt_configuration.password = password;
283#else
284 m_mqtt_configuration.credentials.client_id = client_id;
285 m_mqtt_configuration.credentials.username = user_name;
286 m_mqtt_configuration.credentials.authentication.password = password;
287#endif // ESP_IDF_VERSION_MAJOR < 5
288 // Update configuration is called to ensure that if we connected previously and call connect again with other credentials,
289 // then we also update the client_id, username and password we connect with. Especially important for the provisioning workflow to work correctly
290 update_configuration();
291
292 // Check wheter the client has been initalzed before already, it it has we do not want to reinitalize,
293 // but simply force reconnection with the client because it has lost that connection
294 if (m_mqtt_client != nullptr) {
295 const esp_err_t error = esp_mqtt_client_reconnect(m_mqtt_client);
296 return error == ESP_OK;
297 }
298
299 // The client is first initalized once the connect has actually been called, this is done because the passed setting are required for the client inizialitation structure,
300 // additionally before we attempt to connect with the client we have to ensure it is configued by then.
301 m_mqtt_client = esp_mqtt_client_init(&m_mqtt_configuration);
302 esp_err_t error = esp_mqtt_client_register_event(m_mqtt_client, esp_mqtt_event_id_t::MQTT_EVENT_ANY, Espressif_MQTT_Client::static_mqtt_event_handler, this);
303
304 if (error != ESP_OK) {
305 return false;
306 }
307 error = esp_mqtt_client_start(m_mqtt_client);
308 return error == ESP_OK;
309 }

321 bool publish(char const * topic, uint8_t const * payload, size_t const & length) override {
322 int message_id = MQTT_FAILURE_MESSAGE_ID;
323
324 if (m_enqueue_messages) {
325 message_id = esp_mqtt_client_enqueue(m_mqtt_client, topic, reinterpret_cast<const char*>(payload), length, 0U, 0U, true);
326 return message_id > MQTT_FAILURE_MESSAGE_ID;
327 }
328
329 // The blocking version esp_mqtt_client_publish() it is sent directly from the users task context.
330 // This way is used to send messages to the cloud, because like that no internal buffer has to be used to store the message until it should be sent,
331 // because all messages are sent with QoS level 0. If this is not wanted esp_mqtt_client_enqueue() could be used with store = true,
332 // to ensure the sending is done in the mqtt event context instead of the users task context.
333 // Allows to use the publish method without having to worry about any CPU overhead, so it can even be used in callbacks or high priority tasks, without starving other tasks,
334 // but compared to the other method esp_mqtt_client_enqueue() requires to save the message in the outbox, which increases the memory requirements for the internal buffer size
335 message_id = esp_mqtt_client_publish(m_mqtt_client, topic, reinterpret_cast<const char*>(payload), length, 0U, 0U);
336 return message_id > MQTT_FAILURE_MESSAGE_ID;
337 }

415 void mqtt_event_handler(esp_event_base_t base, esp_mqtt_event_id_t const & event_id, void * event_data) {
416 esp_mqtt_event_handle_t const event = static_cast<esp_mqtt_event_handle_t>(event_data);
417
418#if THINGSBOARD_ENABLE_DEBUG
419 Logger::printfln(RECEIVED_MQTT_EVENT, esp_event_id_to_name(event_id));
420#endif // THINGSBOARD_ENABLE_DEBUG
421 switch (event_id) {
422 case esp_mqtt_event_id_t::MQTT_EVENT_CONNECTED:
423 m_connected = true;
424 m_connected_callback.Call_Callback();
425 break;
426 case esp_mqtt_event_id_t::MQTT_EVENT_DISCONNECTED:
427 m_connected = false;
428 break;
429 case esp_mqtt_event_id_t::MQTT_EVENT_DATA: {
430 // Check wheter the given message has not bee received completly, but instead would be received in multiple chunks,
431 // if it were we discard the message because receiving a message over multiple chunks is currently not supported
432 if (event->data_len != event->total_data_len) {
433 Logger::printfln(MQTT_DATA_EXCEEDS_BUFFER, event->total_data_len, get_receive_buffer_size());
434 break;
435 }
436 // Topic is not null terminated, to fix this issue we copy the topic string.
437 // This overhead is acceptable, because we nearly always copy only a few bytes (around 20), meaning the overhead is insignificant.
438 char topic[event->topic_len + 1] = {};
439 strncpy(topic, event->topic, event->topic_len);
440 m_received_data_callback.Call_Callback(topic, reinterpret_cast<uint8_t*>(event->data), event->data_len);
441 break;
442 }
443 default:
444 // Nothing to do
445 break;
446 }
447 }
